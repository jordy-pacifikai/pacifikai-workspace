chatbot-workflow.json - Configuration n8n 
// Exemple de configuration workflow chatbot
{
  "name": "Moorings Chatbot Navigation",
  "nodes": [
    {
      "type": "n8n-nodes-base.webhook",
      "name": "Chat Webhook",
      "parameters": {
        "path": "moorings-chat",
        "responseMode": "responseNode"
      }
    },
    {
      "type": "n8n-nodes-base.supabase",
      "name": "Vector Search",
      "parameters": {
        "operation": "select",
        "table": "navigation_docs",
        "options": {
          "filter": "embedding.match(query_embedding, 0.7)"
        }
      }
    },
    {
      "type": "@n8n/n8n-nodes-langchain.openAiChat",
      "name": "Claude Sonnet",
      "parameters": {
        "model": "claude-3-5-sonnet-20241022",
        "systemPrompt": "Tu es un conseiller navigation expert..."
      }
    }
  ]
}
